import React, { useEffect } from 'react'
import { BrowserRouter as Router, Routes, Route, Navigate } from 'react-router-dom'
import { Toaster } from 'sonner'
import { AuthProvider } from './hooks/useAuth'
import Layout from './components/Layout'
import ProtectedRoute from './components/ProtectedRoute'
 
// 页面组件
import Login from './pages/Login'
import Home from './pages/Home'
import PurchaseEntry from './pages/PurchaseEntry'
import PurchaseList from './pages/PurchaseList'
import ProductEntry from './pages/ProductEntry'
import ProductList from './pages/ProductList'
import InventoryList from './pages/InventoryList'
import SupplierManagement from './pages/SupplierManagement'
import UserManagement from './pages/UserManagement'
import Settings from './pages/Settings'
import { useAuth } from './hooks/useAuth'
import { useEnhancedNetworkStatus } from './hooks/useEnhancedNetworkStatus'
import { networkAdaptation } from './utils/networkAdaptation'
import { autoReconnectManager } from './utils/autoReconnectManager'
import { ipAutoSwitchManager } from './utils/ipAutoSwitchManager'
 
// 内部应用组件，在AuthProvider内部使用
function AppContent() {
  const { isAuthenticated, isLoading } = useAuth()
  const { quality, addNetworkListener } = useEnhancedNetworkStatus()
 
  // 初始化网络状态监控和自适应配置
  useEffect(() => {
    // 更新网络自适应配置
    networkAdaptation.updateNetworkQuality(quality)
 
    // 监听网络状态变化
    const removeListener = addNetworkListener((event) => {
      console.log('🌐 网络状态变化事件:', event)
      
      // 根据网络状态变化更新配置
      if (event.type === 'quality_change' || event.type === 'online') {
        const currentQuality = event.currentStatus.isOnline 
          ? quality 
          : '离线'
        networkAdaptation.updateNetworkQuality(currentQuality)
      }
      
      // 智能调整配置
      if (event.type === 'quality_change') {
        setTimeout(() => {
          networkAdaptation.smartAdjustConfig()
        }, 1000) // 延迟1秒后调整，避免频繁调整
      }
    })
 
    // 定期智能调整配置（每5分钟）
    const configAdjustInterval = setInterval(() => {
      networkAdaptation.smartAdjustConfig()
    }, 5 * 60 * 1000)
 
    return () => {
      removeListener()
      clearInterval(configAdjustInterval)
    }
  }, [quality, addNetworkListener])
 
  // 集成自动重连管理器
  useEffect(() => {
    // 监听重连事件
    const handleReconnectEvent = (event: any) => {
      switch (event.type) {
        case 'reconnect-start':
          console.log('开始重连...', event.data)
          break
        case 'reconnect-success':
          console.log('重连成功', event.data)
          break
        case 'reconnect-failure':
          console.log('重连失败', event.data)
          break
        case 'sync-start':
          console.log('开始数据同步...')
          break
        case 'sync-success':
          console.log('数据同步完成')
          break
        case 'sync-failure':
          console.log('数据同步失败', event.data)
          break
        default:
          break
      }
    }
 
    autoReconnectManager.addEventListener(handleReconnectEvent)
 
    return () => {
       autoReconnectManager.removeEventListener(handleReconnectEvent)
     }
   }, [])
 
   // 集成IP自动切换管理器
   useEffect(() => {
     // 监听端点切换事件
     const handleEndpointSwitch = (endpoint: any) => {
       console.log('API端点已切换:', endpoint.name, endpoint.url)
     }
 
     // 监听API URL变化事件
     const handleApiUrlChange = (event: CustomEvent) => {
       console.log('API URL已更新:', event.detail.url)
       // 可以在这里触发重新获取数据等操作
     }
 
     ipAutoSwitchManager.addSwitchListener(handleEndpointSwitch)
     window.addEventListener('api-url-changed', handleApiUrlChange as EventListener)
 
     return () => {
       ipAutoSwitchManager.removeSwitchListener(handleEndpointSwitch)
       window.removeEventListener('api-url-changed', handleApiUrlChange as EventListener)
     }
   }, [])
 
  if (isLoading) {
    return (
      <div className="min-h-screen bg-gray-50 flex items-center justify-center">
        <div className="text-center">
          <div className="animate-spin rounded-full h-12 w-12 border-b-2 border-crystal-500 mx-auto mb-4"></div>
          <p className="text-gray-600">加载中...</p>
        </div>
      </div>
    )
  }
 
  return (
    <Router>
      <div className="min-h-screen bg-gray-50">
        <Toaster 
          position="top-center"
          richColors
          closeButton
          duration={3000}
          toastOptions={{
            style: {
              fontSize: '14px',
            },
          }}
        />
        <Routes>
            {/* 公开路由 */}
            <Route path="/login" element={<Login />} />
            
            {/* 受保护的路由 */}
            <Route path="/" element={
              <ProtectedRoute>
                <Layout>
                  <Home />
                </Layout>
              </ProtectedRoute>
            } />
            
            <Route path="/purchase-entry" element={
              <ProtectedRoute>
                <Layout>
                  <PurchaseEntry />
                </Layout>
              </ProtectedRoute>
            } />
            
            <Route path="/purchase-list" element={
              <ProtectedRoute>
                <Layout>
                  <PurchaseList />
                </Layout>
              </ProtectedRoute>
            } />
            
 
            
            <Route path="/product-entry" element={
              <ProtectedRoute>
                <Layout>
                  <ProductEntry />
                </Layout>
              </ProtectedRoute>
            } />
            
            <Route path="/product-list" element={
              <ProtectedRoute>
                <Layout>
                  <ProductList />
                </Layout>
              </ProtectedRoute>
            } />
            
            <Route path="/inventory-list" element={
              <ProtectedRoute>
                <Layout>
                  <InventoryList />
                </Layout>
              </ProtectedRoute>
            } />
            
            <Route path="/supplier-management" element={
              <ProtectedRoute required_role="BOSS">
                <Layout>
                  <SupplierManagement />
                </Layout>
              </ProtectedRoute>
            } />
            
            <Route path="/user-management" element={
              <ProtectedRoute required_role="BOSS">
                <Layout>
                  <UserManagement />
                </Layout>
              </ProtectedRoute>
            } />
            
            <Route path="/settings" element={
              <ProtectedRoute>
                <Layout>
                  <Settings />
                </Layout>
              </ProtectedRoute>
            } />
            
            {/* 默认重定向 */}
            <Route path="*" element={<Navigate to="/" replace />} />
          </Routes>
        </div>
      </Router>
    )
}
 
// 主App组件，包装AuthProvider
function App() {
  return (
    <AuthProvider>
      <AppContent />
    </AuthProvider>
  )
}
 
export default App
