          ...msg,
          timestamp: new Date(msg.timestamp)

        console.error('Failed to parse chat history:', error)

          timestamp: new Date()

                <div className="space-y-3">

                          {message.timestamp.toLocaleTimeString()}
                        </p>
                      </div>
                    </div>
                  ))}
                </div>
              )}
            </div>
            
            {/* 输入区域 */}
            <div className="flex space-x-2">
              <textarea
                value={inputMessage}
                onChange={(e) => setInputMessage(e.target.value)}
                onKeyPress={handleKeyPress}
                className="flex-1 resize-none border border-gray-300 rounded-lg px-3 py-2 text-sm md:text-base focus:outline-none focus:ring-2 focus:ring-crystal-500 focus:border-transparent leading-normal"
                rows={2}
                placeholder="输入您的问题..."
              />
              <button 
                onClick={handleSendMessage}
                className="bg-crystal-500 hover:bg-crystal-600 text-white p-2 rounded-lg transition-colors duration-200"
              >
                <Send className="h-4 w-4" />
              </button>
            </div>
          </div>
        </div>
      </div>
      
      {/* 智能助手 - 移动端悬浮按钮 - 固定在右上角 */}
      <div className="md:hidden">
        <button
          onClick={handleFloatingButtonClick}
          className="fixed top-16 right-4 z-50 w-12 h-12 bg-crystal-500 hover:bg-crystal-600 text-white rounded-full shadow-lg transition-all duration-200 flex items-center justify-center"
        >
          {isChatOpen ? (
            <X className="h-5 w-5" />
          ) : (
            <>
              <Bot className="h-5 w-5" />
              <div className="absolute -top-1 -right-1 w-2.5 h-2.5 bg-green-500 rounded-full border-2 border-white"></div>
            </>
          )}
        </button>
        
        {/* 展开的聊天窗口 */}
        {isChatOpen && (
          <div className="fixed inset-0 z-40 flex items-center justify-center p-4">
            {/* 遮罩层 */}
            <div 
              className="absolute inset-0 bg-black bg-opacity-50"
              onClick={() => setIsChatOpen(false)}
            ></div>
            
            {/* 聊天窗口 - 居中长条形 */}
            <div className="relative w-full max-w-md bg-white rounded-xl shadow-lg animate-slide-up max-h-[80vh] flex flex-col">
              <div className="p-4 border-b border-gray-200 flex-shrink-0">
                <div className="flex items-center justify-between">
                  <div className="flex items-center space-x-2">
                    <Bot className="h-5 w-5 text-crystal-600" />
                    <h3 className="text-base md:text-lg font-semibold text-gray-900 leading-normal">智能助手</h3>
                    <div className="w-2 h-2 bg-green-500 rounded-full"></div>
                  </div>
                  <button
                    onClick={() => setIsChatOpen(false)}
                    className="text-gray-400 hover:text-gray-600"
                  >
                    <X className="h-5 w-5" />
                  </button>
                </div>
              </div>
              
              <div className="flex-1 bg-gray-50 p-4 overflow-y-auto scrollbar-hide min-h-0">
                {messages.length === 0 ? (
                  <div className="bg-white rounded-lg p-3 shadow-sm">
                    <p className="text-sm md:text-base text-gray-700 leading-normal">
                      您好！我是水晶ERP智能助手，可以帮助您解答关于系统使用、数据查询等问题。请问有什么可以帮助您的吗？
                    </p>
                  </div>
                ) : (
                  <div className="space-y-3">
                    {messages.map((message) => (
                      <div
                        key={message.id}
                        className={`flex ${message.isUser ? 'justify-end' : 'justify-start'}`}
                      >
                        <div
                          className={`max-w-[80%] rounded-lg p-3 shadow-sm ${
                            message.isUser
                              ? 'bg-crystal-500 text-white'
                              : 'bg-white text-gray-700'
                          }`}
                        >
                          <p className="text-sm md:text-base whitespace-pre-wrap leading-normal">{message.text}</p>
                          <p className={`text-xs mt-1 leading-normal ${
                            message.isUser ? 'text-crystal-100' : 'text-gray-400'
                          }`}>
                            {message.timestamp.toLocaleTimeString()}
                          </p>
                        </div>
                      </div>
                    ))}
                  </div>
                )}
                <div ref={messagesEndRef} />
              </div>
              
              <div className="p-4 border-t border-gray-200 flex-shrink-0">
                <div className="flex space-x-2">
                  <textarea
                    value={inputMessage}
                    onChange={(e) => setInputMessage(e.target.value)}
                    onKeyPress={handleKeyPress}
                    className="flex-1 resize-none border border-gray-300 rounded-lg px-3 py-2 text-sm md:text-base focus:outline-none focus:ring-2 focus:ring-crystal-500 focus:border-transparent leading-normal"
                    rows={2}
                    placeholder="输入您的问题..."
                  />
                  <button
                    onClick={handleSendMessage}
                    className="bg-crystal-500 hover:bg-crystal-600 text-white p-2 rounded-lg transition-colors duration-200"
                  >
                    <Send className="h-4 w-4" />
                  </button>
                </div>
              </div>
            </div>
          </div>
        )}
      </div>
    </div>
  )
}
