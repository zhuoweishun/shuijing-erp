            <AlertCircle className="h-5 w-5 text-yellow-600 mr-2" />

          </div>
        </div>
      )}
 
      {/* 数据列表 */}
      {isAuthenticated && !state.loading && !state.error && (
        <>
          {state.purchases.length === 0 ? (
            <div className="bg-white rounded-xl shadow-sm border border-gray-200 p-12 text-center">

              >
                <Plus className="h-4 w-4 mr-2" />
                新增采购

            </div>
          ) : (
            <>
              {isMobile ? renderMobileCards() : renderDesktopTable()}
              {renderPagination()}
            </>
          )}
        </>
      )}

      {state.image_preview.isOpen && (
        <div

                 maxWidth: '90vw',
                 maxHeight: '90vh'
               }}
             />
            <button
              onClick={closeImagePreview}
              className="absolute top-2 right-2 sm:top-4 sm:right-4 bg-black bg-opacity-50 text-white p-2 rounded-full hover:bg-opacity-75 transition-all z-10"
              title="关闭"
            >
              <svg className="w-5 h-5 sm:w-6 sm:h-6" fill="none" stroke="current_color" viewBox="0 0 24 24">
                <path strokeLinecap="round" strokeLinejoin="round" strokeWidth={2} d="M6 18L18 6M6 6l12 12" />
              </svg>
            </button>
            <div className="absolute bottom-2 left-2 sm:bottom-4 sm:left-4 bg-black bg-opacity-50 text-white px-2 py-1 sm:px-3 sm:py-2 rounded-lg max-w-[80vw] z-10">
              <p className="text-xs sm:text-sm font-medium truncate">{state.image_preview.alt_text}</p>
            </div>
          </div>
        </div>
      )}
    </div>
  )
