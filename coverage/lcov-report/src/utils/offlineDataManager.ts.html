import { offlineCache } from './offlineCache'
import { toast } from 'sonner'
 
/**
 * 离线数据操作类型
 */
export type OfflineOperationType = 'CREATE' | 'UPDATE' | 'DELETE'
 
/**
 * 离线数据操作接口
 */
export interface OfflineOperation {
  id: string
  type: OfflineOperationType
  endpoint: string
  data: any
  timestamp: number
  retryCount: number
  maxRetries: number
  priority: 'high' | 'medium' | 'low'
  metadata?: {
    entityType?: string
    entityId?: string
    description?: string
  }
}
 
/**
 * 离线数据同步结果
 */
export interface SyncResult {
  success: boolean
  operation: OfflineOperation
  error?: string
  response?: any
}
 
/**
 * 离线数据管理器
 * 负责管理离线状态下的数据操作和同步
 */
class OfflineDataManager {
  private operations: Map<string, OfflineOperation> = new Map()
  private syncInProgress = false
  private syncListeners: Array<(result: SyncResult) => void> = []
  private readonly STORAGE_KEY = 'offline_operations'
  private readonly MAX_OPERATIONS = 1000
 
  constructor() {
    this.loadOperations()
    this.setupEventListeners()
  }
 
  /**
   * 设置事件监听器
   */
  private setupEventListeners() {
    // 监听网络状态变化
    window.addEventListener('online', () => {
      this.handleNetworkReconnect()
    })
 
    // 监听页面卸载，保存操作
    window.addEventListener('beforeunload', () => {
      this.saveOperations()
    })
 
    // 定期保存操作
    setInterval(() => {
      this.saveOperations()
    }, 30000) // 每30秒保存一次
  }
 
  /**
   * 从本地存储加载操作
   */
  private loadOperations() {
    try {
      const stored = localStorage.getItem(this.STORAGE_KEY)
      if (stored) {
        const operations: OfflineOperation[] = JSON.parse(stored)
        operations.forEach(op => {
          this.operations.set(op.id, op)
        })
      }
    } catch (error) {
      console.error('Failed to load offline operations:', error)
    }
  }
 
  /**
   * 保存操作到本地存储
   */
  private saveOperations() {
    try {
      const operations = Array.from(this.operations.values())
      localStorage.setItem(this.STORAGE_KEY, JSON.stringify(operations))
    } catch (error) {
      console.error('Failed to save offline operations:', error)
    }
  }
 
  /**
   * 添加离线操作
   */
  addOperation(
    type: OfflineOperationType,
    endpoint: string,
    data: any,
    options: {
      priority?: 'high' | 'medium' | 'low'
      maxRetries?: number
      metadata?: OfflineOperation['metadata']
    } = {}
  ): string {
    const id = this.generateOperationId()
    const operation: OfflineOperation = {
      id,
      type,
      endpoint,
      data,
      timestamp: Date.now(),
      retryCount: 0,
      maxRetries: options.maxRetries || 3,
      priority: options.priority || 'medium',
      metadata: options.metadata
    }
 
    this.operations.set(id, operation)
    this.cleanupOldOperations()
    this.saveOperations()
 
    // 显示离线操作提示
    this.showOfflineOperationToast(operation)
 
    return id
  }
 
  /**
   * 移除操作
   */
  removeOperation(id: string): boolean {
    const removed = this.operations.delete(id)
    if (removed) {
      this.saveOperations()
    }
    return removed
  }
 
  /**
   * 获取所有待同步操作
   */
  getPendingOperations(): OfflineOperation[] {
    return Array.from(this.operations.values())
      .sort((a, b) => {
        // 按优先级和时间排序
        const priorityOrder = { high: 3, medium: 2, low: 1 }
        const priorityDiff = priorityOrder[b.priority] - priorityOrder[a.priority]
        if (priorityDiff !== 0) return priorityDiff
        return a.timestamp - b.timestamp
      })
  }
 
  /**
   * 获取操作统计信息
   */
  getStats() {
    const operations = Array.from(this.operations.values())
    const byType = operations.reduce((acc, op) => {
      acc[op.type] = (acc[op.type] || 0) + 1
      return acc
    }, {} as Record<OfflineOperationType, number>)
 
    const byPriority = operations.reduce((acc, op) => {
      acc[op.priority] = (acc[op.priority] || 0) + 1
      return acc
    }, {} as Record<string, number>)
 
    return {
      total: operations.length,
      byType,
      byPriority,
      oldestTimestamp: operations.length > 0 ? Math.min(...operations.map(op => op.timestamp)) : 0,
      newestTimestamp: operations.length > 0 ? Math.max(...operations.map(op => op.timestamp)) : 0
    }
  }
 
  /**
   * 同步所有待处理操作
   */
  async syncAll(): Promise<SyncResult[]> {
    if (this.syncInProgress) {
      throw new Error('Sync already in progress')
    }
 
    if (!navigator.onLine) {
      throw new Error('Network is offline')
    }
 
    this.syncInProgress = true
    const results: SyncResult[] = []
    const operations = this.getPendingOperations()
 
    try {
      for (const operation of operations) {
        const result = await this.syncOperation(operation)
        results.push(result)
        
        // 通知监听器
        this.syncListeners.forEach(listener => listener(result))
        
        if (result.success) {
          this.removeOperation(operation.id)
        } else {
          // 增加重试次数
          operation.retryCount++
          if (operation.retryCount >= operation.maxRetries) {
            // 达到最大重试次数，移除操作
            this.removeOperation(operation.id)
            toast.error(`操作同步失败: ${operation.metadata?.description || operation.endpoint}`, {
              description: `已达到最大重试次数 (${operation.maxRetries})`,
              duration: 5000
            })
          }
        }
        
        // 添加延迟避免过于频繁的请求
        await new Promise(resolve => setTimeout(resolve, 100))
      }
    } finally {
      this.syncInProgress = false
      this.saveOperations()
    }
 
    return results
  }
 
  /**
   * 同步单个操作
   */
  private async syncOperation(operation: OfflineOperation): Promise<SyncResult> {
    try {
      const response = await this.executeOperation(operation)
      return {
        success: true,
        operation,
        response
      }
    } catch (error) {
      return {
        success: false,
        operation,
        error: error instanceof Error ? error.message : 'Unknown error'
      }
    }
  }
 
  /**
   * 执行操作
   */
  private async executeOperation(operation: OfflineOperation): Promise<any> {
    const { type, endpoint, data } = operation
    
    const options: RequestInit = {
      method: this.getHttpMethod(type),
      headers: {
        'Content-Type': 'application/json',
      },
      body: type !== 'DELETE' ? JSON.stringify(data) : undefined
    }
 
    const response = await fetch(endpoint, options)
    
    if (!response.ok) {
      throw new Error(`HTTP ${response.status}: ${response.statusText}`)
    }
 
    return response.json()
  }
 
  /**
   * 获取HTTP方法
   */
  private getHttpMethod(type: OfflineOperationType): string {
    switch (type) {
      case 'CREATE': return 'POST'
      case 'UPDATE': return 'PUT'
      case 'DELETE': return 'DELETE'
      default: return 'POST'
    }
  }
 
  /**
   * 处理网络重连
   */
  private async handleNetworkReconnect() {
    if (this.operations.size === 0) return
 
    toast.info('网络已恢复，正在同步离线数据...', {
      duration: 3000
    })
 
    try {
      const results = await this.syncAll()
      const successCount = results.filter(r => r.success).length
      const failCount = results.filter(r => !r.success).length
 
      if (successCount > 0) {
        toast.success(`成功同步 ${successCount} 个操作`, {
          description: failCount > 0 ? `${failCount} 个操作同步失败` : undefined,
          duration: 5000
        })
      }
 
      if (failCount > 0 && successCount === 0) {
        toast.error(`${failCount} 个操作同步失败`, {
          description: '请检查网络连接或稍后重试',
          duration: 5000
        })
      }
    } catch (error) {
      toast.error('数据同步失败', {
        description: error instanceof Error ? error.message : '未知错误',
        duration: 5000
      })
    }
  }
 
  /**
   * 显示离线操作提示
   */
  private showOfflineOperationToast(operation: OfflineOperation) {
    const actionText = {
      CREATE: '创建',
      UPDATE: '更新', 
      DELETE: '删除'
    }[operation.type]
 
    toast.info(`${actionText}操作已保存`, {
      description: `将在网络恢复后自动同步 - ${operation.metadata?.description || operation.endpoint}`,
      duration: 3000
    })
  }
 
  /**
   * 生成操作ID
   */
  private generateOperationId(): string {
    return `op_${Date.now()}_${Math.random().toString(36).substr(2, 9)}`
  }
 
  /**
   * 清理旧操作
   */
  private cleanupOldOperations() {
    if (this.operations.size <= this.MAX_OPERATIONS) return
 
    const operations = Array.from(this.operations.values())
      .sort((a, b) => a.timestamp - b.timestamp)
    
    const toRemove = operations.slice(0, operations.length - this.MAX_OPERATIONS)
    toRemove.forEach(op => this.operations.delete(op.id))
  }
 
  /**
   * 添加同步监听器
   */
  addSyncListener(listener: (result: SyncResult) => void) {
    this.syncListeners.push(listener)
  }
 
  /**
   * 移除同步监听器
   */
  removeSyncListener(listener: (result: SyncResult) => void) {
    const index = this.syncListeners.indexOf(listener)
    if (index > -1) {
      this.syncListeners.splice(index, 1)
    }
  }
 
  /**
   * 清除所有操作
   */
  clear() {
    this.operations.clear()
    this.saveOperations()
  }
 
  /**
   * 强制同步
   */
  async forceSync(): Promise<SyncResult[]> {
    return this.syncAll()
  }
 
  /**
   * 检查是否有待同步操作
   */
  hasPendingOperations(): boolean {
    return this.operations.size > 0
  }
 
  /**
   * 获取同步状态
   */
  isSyncInProgress(): boolean {
    return this.syncInProgress
  }
}
 
// 创建全局实例
export const offlineDataManager = new OfflineDataManager()
 
// 导出类型和实例
export default offlineDataManager
