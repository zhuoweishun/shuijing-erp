/**
 * 网络质量自适应配置管理器
 * 根据网络质量自动调整API请求策略
 */
 
import { useEnhancedNetworkStatus } from '../hooks/useEnhancedNetworkStatus'
 
// 网络质量等级
type NetworkQuality = '优秀' | '良好' | '一般' | '较差' | '离线'
 
// 请求配置
interface RequestConfig {
  timeout: number // 超时时间（毫秒）
  maxRetries: number // 最大重试次数
  retryDelay: number // 重试延迟（毫秒）
  batchSize: number // 批量请求大小
  concurrentRequests: number // 并发请求数
  compressionEnabled: boolean // 是否启用压缩
  cacheStrategy: 'aggressive' | 'normal' | 'minimal' // 缓存策略
}
 
// 网络质量配置映射
const NETWORK_QUALITY_CONFIGS: Record<NetworkQuality, RequestConfig> = {
  '优秀': {
    timeout: 10000, // 10秒
    maxRetries: 2,
    retryDelay: 1000, // 1秒
    batchSize: 50,
    concurrentRequests: 6,
    compressionEnabled: false,
    cacheStrategy: 'normal'
  },
  '良好': {
    timeout: 15000, // 15秒
    maxRetries: 3,
    retryDelay: 1500, // 1.5秒
    batchSize: 30,
    concurrentRequests: 4,
    compressionEnabled: true,
    cacheStrategy: 'normal'
  },
  '一般': {
    timeout: 20000, // 20秒
    maxRetries: 4,
    retryDelay: 2000, // 2秒
    batchSize: 20,
    concurrentRequests: 2,
    compressionEnabled: true,
    cacheStrategy: 'aggressive'
  },
  '较差': {
    timeout: 30000, // 30秒
    maxRetries: 5,
    retryDelay: 3000, // 3秒
    batchSize: 10,
    concurrentRequests: 1,
    compressionEnabled: true,
    cacheStrategy: 'aggressive'
  },
  '离线': {
    timeout: 5000, // 5秒（快速失败）
    maxRetries: 0,
    retryDelay: 0,
    batchSize: 1,
    concurrentRequests: 0,
    compressionEnabled: false,
    cacheStrategy: 'aggressive'
  }
}
 
// 请求优先级
type RequestPriority = 'high' | 'normal' | 'low'
 
// 优先级配置调整
const PRIORITY_ADJUSTMENTS: Record<RequestPriority, Partial<RequestConfig>> = {
  high: {
    timeout: 1.5, // 乘数
    maxRetries: 1.5,
    retryDelay: 0.8
  },
  normal: {
    // 使用默认配置
  },
  low: {
    timeout: 0.7,
    maxRetries: 0.5,
    retryDelay: 1.5
  }
}
 
class NetworkAdaptationManager {
  private currentConfig: RequestConfig
  private currentQuality: NetworkQuality = '良好'
  private configHistory: Array<{ quality: NetworkQuality; timestamp: number }> = []
  private listeners: Array<(config: RequestConfig) => void> = []
 
  constructor() {
    this.currentConfig = NETWORK_QUALITY_CONFIGS['良好']
    this.loadConfigHistory()
  }
 
  /**
   * 根据网络质量获取请求配置
   */
  getRequestConfig(priority: RequestPriority = 'normal'): RequestConfig {
    const baseConfig = { ...this.currentConfig }
    const adjustments = PRIORITY_ADJUSTMENTS[priority]
 
    // 应用优先级调整
    if (adjustments.timeout && typeof adjustments.timeout === 'number') {
      baseConfig.timeout = Math.round(baseConfig.timeout * adjustments.timeout)
    }
    
    if (adjustments.maxRetries && typeof adjustments.maxRetries === 'number') {
      baseConfig.maxRetries = Math.max(0, Math.round(baseConfig.maxRetries * adjustments.maxRetries))
    }
    
    if (adjustments.retryDelay && typeof adjustments.retryDelay === 'number') {
      baseConfig.retryDelay = Math.round(baseConfig.retryDelay * adjustments.retryDelay)
    }
 
    return baseConfig
  }
 
  /**
   * 更新网络质量
   */
  updateNetworkQuality(quality: NetworkQuality): void {
    if (this.currentQuality === quality) return
 
    console.log(`🔄 网络质量变化: ${this.currentQuality} → ${quality}`)
    
    this.currentQuality = quality
    this.currentConfig = NETWORK_QUALITY_CONFIGS[quality]
    
    // 记录配置历史
    this.configHistory.push({
      quality,
      timestamp: Date.now()
    })
    
    // 保持历史记录在合理范围内（最近100条）
    if (this.configHistory.length > 100) {
      this.configHistory = this.configHistory.slice(-100)
    }
    
    this.saveConfigHistory()
    this.notifyListeners()
    
    console.log('📊 当前网络配置:', this.currentConfig)
  }
 
  /**
   * 获取当前网络质量
   */
  getCurrentQuality(): NetworkQuality {
    return this.currentQuality
  }
 
  /**
   * 获取当前配置
   */
  getCurrentConfig(): RequestConfig {
    return { ...this.currentConfig }
  }
 
  /**
   * 添加配置变化监听器
   */
  addConfigListener(listener: (config: RequestConfig) => void): () => void {
    this.listeners.push(listener)
    
    // 返回移除监听器的函数
    return () => {
      const index = this.listeners.indexOf(listener)
      if (index > -1) {
        this.listeners.splice(index, 1)
      }
    }
  }
 
  /**
   * 通知所有监听器
   */
  private notifyListeners(): void {
    this.listeners.forEach(listener => {
      try {
        listener(this.currentConfig)
      } catch (error) {
        console.error('网络配置监听器执行失败:', error)
      }
    })
  }
 
  /**
   * 获取网络质量趋势
   */
  getQualityTrend(minutes: number = 10): {
    trend: 'improving' | 'stable' | 'degrading'
    averageQuality: number
    changes: number
  } {
    const cutoffTime = Date.now() - (minutes * 60 * 1000)
    const recentHistory = this.configHistory.filter(h => h.timestamp >= cutoffTime)
    
    if (recentHistory.length < 2) {
      return {
        trend: 'stable',
        averageQuality: this.getQualityScore(this.currentQuality),
        changes: 0
      }
    }
    
    const qualityScores = recentHistory.map(h => this.getQualityScore(h.quality))
    const averageQuality = qualityScores.reduce((sum, score) => sum + score, 0) / qualityScores.length
    
    const firstScore = qualityScores[0]
    const lastScore = qualityScores[qualityScores.length - 1]
    
    let trend: 'improving' | 'stable' | 'degrading'
    if (lastScore > firstScore + 0.5) {
      trend = 'improving'
    } else if (lastScore < firstScore - 0.5) {
      trend = 'degrading'
    } else {
      trend = 'stable'
    }
    
    return {
      trend,
      averageQuality,
      changes: recentHistory.length - 1
    }
  }
 
  /**
   * 获取质量分数（用于趋势分析）
   */
  private getQualityScore(quality: NetworkQuality): number {
    switch (quality) {
      case '优秀': return 5
      case '良好': return 4
      case '一般': return 3
      case '较差': return 2
      case '离线': return 1
      default: return 3
    }
  }
 
  /**
   * 智能调整配置
   * 基于历史数据和当前趋势进行微调
   */
  smartAdjustConfig(): void {
    const trend = this.getQualityTrend(5) // 分析最近5分钟的趋势
    const baseConfig = NETWORK_QUALITY_CONFIGS[this.currentQuality]
    
    // 根据趋势进行微调
    if (trend.trend === 'degrading') {
      // 网络质量下降，增加超时时间和重试次数
      this.currentConfig = {
        ...baseConfig,
        timeout: Math.round(baseConfig.timeout * 1.2),
        maxRetries: Math.min(baseConfig.maxRetries + 1, 6),
        retryDelay: Math.round(baseConfig.retryDelay * 1.3)
      }
      console.log('📉 检测到网络质量下降，已调整配置为更保守策略')
    } else if (trend.trend === 'improving') {
      // 网络质量改善，可以稍微激进一些
      this.currentConfig = {
        ...baseConfig,
        timeout: Math.round(baseConfig.timeout * 0.9),
        maxRetries: Math.max(baseConfig.maxRetries - 1, 1),
        retryDelay: Math.round(baseConfig.retryDelay * 0.8)
      }
      console.log('📈 检测到网络质量改善，已调整配置为更激进策略')
    } else {
      // 网络质量稳定，使用标准配置
      this.currentConfig = { ...baseConfig }
    }
    
    this.notifyListeners()
  }
 
  /**
   * 保存配置历史到localStorage
   */
  private saveConfigHistory(): void {
    try {
      localStorage.setItem('network_config_history', JSON.stringify(this.configHistory))
    } catch (error) {
      console.warn('保存网络配置历史失败:', error)
    }
  }
 
  /**
   * 从localStorage加载配置历史
   */
  private loadConfigHistory(): void {
    try {
      const saved = localStorage.getItem('network_config_history')
      if (saved) {
        this.configHistory = JSON.parse(saved)
      }
    } catch (error) {
      console.warn('加载网络配置历史失败:', error)
      this.configHistory = []
    }
  }
 
  /**
   * 重置配置为默认值
   */
  resetToDefault(): void {
    this.currentQuality = '良好'
    this.currentConfig = NETWORK_QUALITY_CONFIGS['良好']
    this.configHistory = []
    this.saveConfigHistory()
    this.notifyListeners()
    console.log('🔄 网络配置已重置为默认值')
  }
 
  /**
   * 获取配置统计信息
   */
  getConfigStats(): {
    currentQuality: NetworkQuality
    configChanges24h: number
    averageQuality24h: number
    mostCommonQuality: NetworkQuality
  } {
    const last24h = Date.now() - (24 * 60 * 60 * 1000)
    const recent = this.configHistory.filter(h => h.timestamp >= last24h)
    
    const qualityCount: Record<NetworkQuality, number> = {
      '优秀': 0,
      '良好': 0,
      '一般': 0,
      '较差': 0,
      '离线': 0
    }
    
    let totalScore = 0
    recent.forEach(h => {
      qualityCount[h.quality]++
      totalScore += this.getQualityScore(h.quality)
    })
    
    const mostCommonQuality = Object.entries(qualityCount)
      .reduce((max, [quality, count]) => 
        count > qualityCount[max as NetworkQuality] ? quality as NetworkQuality : max, 
        '良好' as NetworkQuality
      )
    
    return {
      currentQuality: this.currentQuality,
      configChanges24h: recent.length,
      averageQuality24h: recent.length > 0 ? totalScore / recent.length : 4,
      mostCommonQuality
    }
  }
}
 
// 创建全局网络自适应管理器实例
export const networkAdaptation = new NetworkAdaptationManager()
 
// 导出类型
export type { NetworkQuality, RequestConfig, RequestPriority }
export { NetworkAdaptationManager, NETWORK_QUALITY_CONFIGS }
